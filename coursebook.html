var last_clip_catbook = 'clip-catbook_edit';
var last_clip_coursebook = 'clip-section';



function open_subrow(event, xparent_row_id, clip_type, data_id, toggle, subaction, row_class)
{
var parent_row_id = $(event.currentTarget).closest('tr').attr('id');

// get div ids
child_id = parent_row_id + 'child';
childcontent_id = parent_row_id + 'childcontent';
control_icon_id = parent_row_id + '-i';

// switch clip type to last used
if (row_class == 'expandedrow catbook')
{
if (clip_type == 'last')
{
clip_type = last_clip_catbook;
}
else
{
last_clip_catbook = clip_type;
}
}
else
{
if (clip_type == 'last')
{
clip_type = last_clip_coursebook;
}
else
{
last_clip_coursebook = clip_type;
}
}

// check if a child row exists, if not - create one
if ( $('#'+child_id).length === 0 )
{
// close all other expanded rows
if (!event.shiftKey)
{
$('.'+row_class).remove();
}
// insert child row below the parent_row
rows_in_parent = $('#'+parent_row_id).prop("cells").length;
$('#'+parent_row_id).after('<tr id="' + child_id + '" class="' + row_class + '"><td colspan="' + rows_in_parent + '" id="' +childcontent_id + '"><p align="center"><img src="https://d2jm41oajq77sf.cloudfront.net/images/ajax-loader.gif" /></p></td></tr>');
// change +/- control to minus
$('#'+control_icon_id).attr({ src: "https://d2jm41oajq77sf.cloudfront.net/images/foldicon-minus.gif"});

// load clip file into child
$('#'+childcontent_id).load('https://coursebook.utdallas.edu/clips/'+clip_type+'.zog', { id: data_id, div: childcontent_id, subaction: subaction, visit: 'no robots allowed.', hi: 'You agree that you will not use any robot, spider, scraper or other automated means to access this website for any purpose without our prior, express written permission.'} );
}

// if toggle is ON and row exists - remove the child row (i.e. toggle its display)
else if (toggle)
{
// remove child row
$('#'+child_id).remove();
// change +/- control to plus
$('#'+control_icon_id).attr({ src: "https://d2jm41oajq77sf.cloudfront.net/images/foldicon-plus.gif"});
}

// if toggle is OFF and row exists - reload the child row
else
{
// load clip file into child
$('#'+childcontent_id).load('https://coursebook.utdallas.edu/clips/'+clip_type+'.zog', { id: data_id, div: childcontent_id, subaction: subaction, hi: 'hello sailor' } ).hide().fadeIn('slow');
}

// scroll to subrow
var scrollto=$('#'+childcontent_id).offset().top;
$('body,html,document').animate({ scrollTop: scrollto }, 1000);

}
function dosearch(srch_type)
{
var srch = $('#srch').val().replace(/[^\w\-\.\!\: ]/gi, '');

// advanced search - add items from selection lists
if (srch_type == 'adv')
{
$.each( $('.selectlist-list'), function(i, o) { v = $(o).val(); if (v != null) { srch += ' ' + v; }; } );
//$('.selectlist-list').each( function() { alert(this.val()); } );
}
// catbook search
else if (srch_type == 'cat')
{
// check for commands
var match = /(cmd_)([\w]*) ?([\w]*)?/.exec(srch);
if (match != null)
{
alertify.log(match[2]+" command received");
do_cmd(match[2], match[3]);
exit;
}

srch += ' ' + $('#y').val();
$.each( $('.options'), function(i, o) { if ( $(o).is(':checked') ) { srch += ' ' + $(o).attr('id') } } );
}
// standard coursebook search
else
{
srch += ' ' + $('#srch_term').val()
if ( $('#has_syllabus').is(':checked')==true ) srch += ' hassyllabus_1';
if ( $('#has_evaluation').is(':checked')==true ) srch += ' hasevaluation_1';
if ( $('#include_independent_study').is(':checked')==true ) srch += ' includeindstudy_1';
}
// substitute token words
srch = srch.replace(/^\s+|\s+$/g,'');
srch = srch.replace(/[, ]+/gi, '/');
srch = srch.replace(/benchmark/gi, 'kw_bm');
srch = srch.replace(/union/gi, 'kw_un');
srch = srch.replace(/select/gi, 'kw_sel');
srch = srch.replace(/insert/gi, 'kw_ins');
srch = srch.replace(/declare/gi, 'kw_dec');
srch = srch.replace(/drop/gi, 'kw_drp');
srch = srch.replace(/update/gi, 'kw_upd');
srch = srch.replace(/system/gi, 'kw_sys');
srch = srch.replace(/database/gi, 'kw_db');
srch = srch.replace(/data/gi, 'kw_dt');
// set url
if (srch_type == 'adv')
{
var srch_url = 'https://coursebook.utdallas.edu/' + srch + '#searchresults';
}
else
{
var srch_url = 'https://coursebook.utdallas.edu/' + srch;
}

// setup proxy redirect
$('#proxysearch').attr('action', srch_url);
$('#proxysearch').submit();
}

// get grade info. a = section_address, n = notify
function peek_grade(a, n)
{
$.getJSON(
'https://coursebook.utdallas.edu/clips/clip-gradepeek.zog',
{'a':a, 'n':n},
function(data)
{
if (data.type == 'prompt')
{
// alertify.set({ buttonFocus: "ok", labels: { ok: "Notify Me", cancel: "Cancel"} } );
alertify.prompt(
data.response,
data.default_value,
function (e, n) {
if (e) { peek_grade(a, n) }
else { console.log('x') }
}
);
}
else if (data.reponse_type == 'notify')
{
// alertify.set({ buttonFocus: "ok", labels: { ok: "OK" } } );
alertify.alert(data.response);
}
else
{
// alertify.set({ buttonFocus: "ok", labels: { ok: "OK" } } );
alertify.alert(data.response);
}
});
}

{

});
